<script type="module">
    // --- TESTING CONFIG ---
    const API_KEY = "AIzaSyAiULQ6D6HoLVMYjf45jGmgeiuEzpV8eMc";
    const API_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${API_KEY}`;

    const recordBtn = document.getElementById('start-record');
    const questionInput = document.getElementById('userQuestion');
    const statusMsg = document.getElementById('voice-status');
    const readingText = document.getElementById('reading-text');

    // 1. Voice-to-Text (Speech Recognition)
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    if (SpeechRecognition) {
        const recognition = new SpeechRecognition();

        recordBtn.onclick = () => {
            recognition.start();
            recordBtn.classList.add('mic-active');
            statusMsg.innerText = "Listening to the universe...";
        };

        recognition.onresult = (event) => {
            const transcript = event.results[0][0].transcript;
            questionInput.value = transcript; // Put voice result into textarea
            recordBtn.classList.remove('mic-active');
            statusMsg.innerText = "";
        };

        recognition.onerror = () => {
            recordBtn.classList.remove('mic-active');
            statusMsg.innerText = "Error capturing voice.";
        };
    }

    // 2. Text-to-Speech (Speak Option)
    document.getElementById('speak-reading').onclick = () => {
        const utterance = new SpeechSynthesisUtterance(readingText.innerText);
        utterance.rate = 0.9; // Slightly slower for a more mystical vibe
        utterance.pitch = 1;
        window.speechSynthesis.speak(utterance);
    };

    // 3. AI Fetch Logic
    document.getElementById('get-reading').onclick = async () => {
        const submitBtn = document.getElementById('get-reading');
        const readingText = document.getElementById('reading-text');

        submitBtn.disabled = true;
        submitBtn.innerText = "Aligning Planets...";

        const promptText = `User: ${document.getElementById('userName').value}, 
                        Location: ${document.getElementById('userLoc').value}, 
                        Question: ${document.getElementById('userQuestion').value}`;

        const finalPrompt = `
        You are a professional, mystical astrologer having a 1-on-1 conversation with a client. 
        Based on these details: ${promptText}
        
        RULES:
        1. Speak directly to the user (e.g. "I see great things for you...").
        2. Do NOT use any Markdown (No #, No *, No **).
        3. Use only plain text and natural paragraphs.
        4. Provide a warm, insightful, and conversational reading.
    `;

        try {
            const response = await fetch(API_URL, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    contents: [{
                        parts: [{ text: finalPrompt }]
                    }]
                })
            });

            if (!response.ok) {
                // This captures the exact error message from Google (e.g., "Invalid API Key")
                const errorData = await response.json();
                console.error("Google API Error:", errorData);
                throw new Error(errorData.error?.message || "Unknown API Error");
            }

            const data = await response.json();
            const result = data.candidates[0].content.parts[0].text;

            document.getElementById('output').classList.remove('d-none');
            readingText.innerText = result;

        } catch (error) {
            console.error("Fetch Error:", error);
            alert("Connection Failed: " + error.message);
        } finally {
            submitBtn.disabled = false;
            submitBtn.innerText = "Consult the Stars";
        }
    };
</script>
